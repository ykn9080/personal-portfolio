from pyspark import SparkContext

# From Mysql table
df_mysql = spark.read.format("jdbc")

   .option("driver", "com.mysql.jdbc.Driver")
   .option("dbtable", "tablename")
   .option("user", "user")
   .option("password", "password")
   .load()
